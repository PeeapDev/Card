V0 Integration Code

Copy this code into your v0 project:

// Step 1: Add the V0-compatible SDK script
<script src="https://checkout.peeap.com/embed/peeap-v0.js"></script>

<script>
// Step 2: Initialize with your public key
PeeapV0.init({
  publicKey: 'pk_live_your_key_here',

  // Called when payment succeeds
  onSuccess: function(payment) {
    console.log('Payment successful!', payment);
    // Redirect to success page or update UI
  },

  // Called on error
  onError: function(error) {
    console.error('Payment failed:', error.message);
    alert('Payment failed: ' + error.message);
  }
});

// Step 3: Create payment on button click
function handlePayment() {
  PeeapV0.pay({
    amount: 500,           // Le 500.00
    currency: 'SLE',
    description: 'School Fees',
    reference: 'order_' + Date.now()
  });
}
</script>
