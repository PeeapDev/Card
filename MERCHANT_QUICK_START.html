‚ö° Method 3: JavaScript (Dynamic Amounts)

Perfect for cart checkouts where the amount changes.

üìã Copy This Code:

<!-- Add this button to your page -->
<button id="pay-button"
        style="background:#4F46E5;color:white;padding:15px 30px;border:none;border-radius:8px;font-weight:600;cursor:pointer;font-size:16px;">
  Pay Now
</button>

<script>
document.getElementById('pay-button').addEventListener('click', function() {
  // Get your dynamic amount (e.g., from cart total)
  var amount = 50000; // Le 500.00
  var description = 'Order #12345';

  // Build checkout URL
  var checkoutUrl = 'https://my.peeap.com/checkout/quick?' +
    'merchant_id=f0c164b7-b608-4ca3-88d9-bc576d765160' +
    '&amount=' + amount +
    '¤cy=SLE' +
    '&description=' + encodeURIComponent(description) +
    '&merchant_name=School%20District' +
    '&success_url=' + encodeURIComponent(window.location.origin + '/success') +
    '&cancel_url=' + encodeURIComponent(window.location.origin + '/cancel');

  // Redirect to checkout
  window.location.href = checkoutUrl;
});
</script>
